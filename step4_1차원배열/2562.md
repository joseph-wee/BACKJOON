# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().split("\n")

let max = parseInt(input[0])
let num = 1;

for (i=1; i<9; i++) {
  if (max < parseInt(input[i])) {
    max = parseInt(input[i])
    num = i+1
  }
}

console.log(max)
console.log(num)
```
