# 문제 1546


# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let N = Number(input[0])  // 과목의 개수
let gradeData = input[1].split(" ").map((i) => {return Number(i)})  // 과목별 점수
let max = 0  // 최댓값
let sum = 0  // 합

/* 과목중 최댓값 계산 */
gradeData.forEach((i) => {
  if (max < i) {
    max = i
  }
})
/* 조작한 점수 */
gradeData = gradeData.map((i) => {
  return i / max * 100
})
/* 조작한 점수의 합 계산 */
gradeData.forEach((i) => {
  sum += i
})

console.log(sum / N)  // 조작한 점수의 합의 평균 출력
```
