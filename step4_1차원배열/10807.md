# 문제 10807

# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let array = input[1].split(" ").map(Number) // 주어진 배열
let v = Number(input[2])  // 찾고자하는 정수
let count = 0 // 같은 정수의 개수

array.forEach((i) =>{
  if (i == v) {
    count += 1
  }
})

console.log(count)
```
