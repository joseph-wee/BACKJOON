# 문제 3052


# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let array = input.map((i) => {
  return parseInt(i) % 42
})

array.forEach((i, j) => {
  array.forEach((k, o) => {
    if ( j != o && i == k) {
      array[o] = null
    }
  })
})

array = array.filter((i) => {
  return i != null
})


console.log(array.length)
```
