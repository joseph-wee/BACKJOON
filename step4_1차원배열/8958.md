# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let T = Number(input[0]) // 테스트 케이스
let data = []  // O,X퀴즈 결과 배열
let sum = 0  // 합
let point = 0;  // 문제의 점수

for (i = 1; i <= T; i++) {
  data = [...input[i]]
  data.forEach((i) => {
    if(i == "O") {
      point++
      sum += point
    }
    else {
      point = 0
    }
  })
  console.log(sum)
  sum = 0
  point = 0
}
```
