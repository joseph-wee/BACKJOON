# 풀이
```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let answer = []

input.forEach((i, j) => {
  if( j == input.length ) {
    return 
  }
  answer.push(parseInt(i.split(" ")[0]) + parseInt(i.split(" ")[1]))
})

  console.log(answer.join("\n"))
```
