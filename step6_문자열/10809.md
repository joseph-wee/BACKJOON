# 풀이
```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("")

let result = []

for (let i = 97; i <= 122; i++) {
  result.push(input.indexOf(String.fromCharCode(i)));
}

console.log(result.join(" "))
```
