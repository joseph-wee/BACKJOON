# 문제 2908

# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split(" ").map(Number)

let A = ""
let B = ""
let array = []

array = input[0].toString().split("")
  for(i = array.length-1; i >= 0; i--){
    A += array[i]
  }

array = input[1].toString().split("")
  for(i = array.length-1; i >= 0; i--){
    B += array[i]
  }

if(Number(A) > Number(B)){
  console.log(A)
}
else {
  console.log(B)
}
```
