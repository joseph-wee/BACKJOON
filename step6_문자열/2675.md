```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let T = Number(input[0])
  let result = ""

for(i = 1; i <= T; i++) {
  let R = Number(input[i].split(" ")[0])
  let S = input[i].split(" ")[1].split("")
  S.forEach((j) => {
    for (k=0; k<R; k++) {
    result += j
    }
  })
  console.log(result)
  result = ""
}
```
