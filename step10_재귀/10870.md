# 문제 10870

# 풀이
```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString()


function fibonachi(num) {
  if (num == 0) {
    return 0
  }
  else if (num == 1) {
    return 1
  }
  else {
    return fibonachi(num - 1) + fibonachi(num - 2)
  }
}

console.log(fibonachi(input))
```
