# 문제 10872

# 풀이
```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString()


function factorial(num) {
  if (num <= 1) {
    return 1
  }
  else {
    return num * factorial(num-1)
  }
}

console.log(factorial(input))
```
