# 문제 2798


# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let M = parseInt(input[0].split(" ")[1])  // M
let card = input[1].split(" ").map(num => parseInt(num))  // 주어진 카드의 개수
let max = 0  // M을 넘지않는 카드 3개를 합친 최대값
let sum = 0  // 카드 3개의 합

card.forEach((i) => {
  card.forEach((j) => {
    card.forEach((k) => {
      if(i != j && i != k && j != k) {
        sum = i + j + k

        if(sum > max && sum <= M) {
          max = sum
        }
      }
    })
  })
})

console.log(max)
```
