# 문제 2231


# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("input.txt").toString().trim()

/*분해합 리턴하는 함수*/
const d = (n) => {
  let sum = parseInt(n)
  n.toString().split("").forEach((i) => {
    sum += parseInt(i)
  })
  return sum
}

let result = [] // 생성자 들어가는 배열

/*생성자를 배열에 저장*/
for(i=1; i<=parseInt(input); i++) {
  if( d(i) == parseInt(input)) {
    result.push(i)
  }
}

let min = 0

/*배열의 길이가 0 이면 생성자가 없으므로 0출력 아니면 생성자들 중 제일 작은 생성자 출력*/
if (result.length == 0) {
  console.log(0)
}
else {
  min = result[0]
  result.forEach((i) => {
    if (min > i) {
      min = i
    }
  })
  console.log(min)
}

```
