# 문제 11050

# 풀이 

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split(" ").map(num => parseInt(num))

n = input[0]  // 자연수 N
k = input[1]  // 정수 K
let answer = 1  // 답

for(i=n; i>1; i--) {
    answer = answer * i
  }
for(i=k; i>1; i--) {
    answer = answer / i
  }
for(i=n-k; i>1; i--) {
    answer = answer / i
  }

console.log(answer)
```
