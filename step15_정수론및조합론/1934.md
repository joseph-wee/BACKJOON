# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let T = parseInt(input[0])
let A = []
let B = []
let numbers = []
let numA = 0
let numB = 0
let multiple = 1

input.shift()
input.forEach((i) => {
  numbers = i.split(" ").map(num => parseInt(num))
  A.push(numbers[0])
  B.push(numbers[1])
})

for (i = 0; i < T; i++) {
  j = 2
  numA = A[i]
  numB = B[i]
  while(j <= 45000) {
    if (numA == 1 || numB == 1 || A[i] == j || B[i] == j) {
      console.log(multiple * numA * numB)
      break
    }
    if (numA % j == 0 && numB % j == 0) {
      numA = numA / j
      numB = numB / j
      multiple = multiple * j
    }
    else {
      j++
    }
  }
  multiple = 1
}
```
