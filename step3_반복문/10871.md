# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().split("\n")

let N = parseInt(input[0].split(" ")[0])
let X = parseInt(input[0].split(" ")[1])
let A = input[1].split(" ")
let answer = []

for (i=0; i<N; i++) {
  if(parseInt(A[i]) < X) {
    answer.push(A[i])
  }
}

console.log(answer.join(" "))
```
