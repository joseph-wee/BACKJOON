# 문제 번호 15552

# 풀이

```javascript
let fs = require('fs');
let input = fs.readFileSync('dev/stdin').toString().split('\n')

let T = input[0]  // 테스트케이스의 개수
let answer = "";

for(i = 1; i <= T; i++) {
  let A = parseInt(input[i].split(" ")[0])
  let B = parseInt(input[i].split(" ")[1])

  answer += A + B + "\n"
}

console.log(answer)
```
