# 문제 11022



# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().split("\n")

let T = input[0]  // 테스트 케이스의 개수

for (i=1; i<=T; i++) {
  let A = parseInt(input[i].split(" ")[0])
  let B = parseInt(input[i].split(" ")[1])

  let sum = A + B

  console.log("Case #" + i + ": " + A + " + " + B + " = "+sum)
}
```
