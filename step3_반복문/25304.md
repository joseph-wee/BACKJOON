# 풀이

```javascript
let fs = require('fs');
let input = fs.readFileSync('dev/stdin').toString().split('\n')

let X = input[0]  // 영수증에 적힌 총 금액
let N = input[1]  // 영수증에 적인 구매한 물건 종류의 수

let a = [];  // 물건의 가격
let b = [];  // 각 물건의 개수

let sum = 0;  // 모든 물건의 합

input.forEach((i, j) => {
  if( j > 1) {
    a.push(i.split(" ")[0])
    b.push(i.split(" ")[1])
  }
})

for (i=0; i<N; i++) {
  sum += a[i] * b[i]
}

if ( X == sum ) {
  console.log("Yes")
}
else {
  console.log("No")
}
```
