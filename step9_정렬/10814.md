## 1차 - 실패

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

let n = input.shift()
let arr = []
let result = ""
let x = 0

for(i=0; i<n; i++) {
  arr.push(input[i].split(" "))
}
let min 
while(arr.length > 0) {
  min = 201
  for(i=0; i<arr.length; i++) {
    if(parseInt(min) > parseInt(arr[i][0])) {
      min = arr[i]
      x = i
    }
  }
  result += min.join(" ") + "\n"
  arr.splice(x, 1)
}

console.log(result)
```
