# 문제 2750



# 풀이
```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n").map(i => {return parseInt(i)})

input.shift()

while( input.length > 0) {
  let min = input[0]
  input.forEach((i) => {
    if(min > i) {
      min = i
    }
  })
  input = input.filter((i) => {
    return i != min
  })
  console.log(min)
}
```
