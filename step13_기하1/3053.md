# 풀이

```javascript
let fs = require("fs");
let R = fs
  .readFileSync("input.txt")
  .toString()
  .trim()
  .split("\n")
  .map((num) => parseInt(num)); // 반지름 R

let result = []; // 결과

/** 유클리드 기하학에서 원의 넓이 */
result.push((Math.PI * R * R).toFixed(6));

/** 택시 기하학에서 원의 넓이 */
result.push((R * R * 2).toFixed(6));

console.log(result.join("\n"));
```
