# 문제 1085



# 풀이

```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split(" ").map(Number)

let min = 10000

if( min > input[0]) {
  min = input[0]
}
if( min > input[2] - input[0]) {
  min = input[2] - input[0]
}
if( min > input[1]) {
  min = input[1]
}
if( min > input[3] - input[1]) {
  min = input[3] - input[1]
}

console.log(min)
```
