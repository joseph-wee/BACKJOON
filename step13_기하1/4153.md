# 문제 4153


# 풀이
```javascript
let fs = require("fs");
let input = fs.readFileSync("dev/stdin").toString().trim().split("\n")

input.pop()

input.forEach((i) => {
  i = i.split(" ").map(Number)
  let max = i[0]
  i.forEach((j) => {
    if(max < j) {
      max = j
    }
  })

  i = i.filter((j) => {
    return j != max
  })

  let a = i[0]
  let b = i[1]
  
  if( a*a + b*b == max*max) {
    console.log("right")
  }
  else {
    console.log("wrong")
  }
})
```
